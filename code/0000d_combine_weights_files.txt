library(dplyr)
library(readr)
library(purrr)

# List all downloaded files matching the pattern *_PRS_weights.txt
weight_files <- list.files(pattern = "^(EUR|AFR)_prscsx_weights\\.txt$")

# Function to read and process each file
read_and_process <- function(file) {
  df <- read_delim(file, delim = " ")  # Adjust delimiter if needed
  
  # Extract the prefix from the filename before _PRS_weights.txt
  prefix <- sub("_PRS_weights.txt", "", basename(file))
  
  # Add a new column for the prefix
  df <- df %>% mutate(disease = prefix)
  
  return(df)
}

# Read and combine all files into one dataframe
combined_df <- map_dfr(weight_files, read_and_process)

# Save combined file
destination_filename <- "combined_weights.txt"
write_delim(combined_df, destination_filename, delim = " ", quote = "none")

# Upload the processed file to the bucket
system(paste0("gsutil cp ./", destination_filename, " ", my_bucket, "/weights/hg38/"), intern=T)
